Q-32: Write a C/C++ program to find the position of given element in an array. 
    
Using - C

  

  

    
#include<stdio.h>

void main() {
	int size, element, i;
	printf("Enter the size of the Array:  ");
	scanf("%d",&size);
	int arr[size];
	printf("Enter %d Elements:  \n");
	for(i=0; i < size; i++) {
		scanf("%d", &arr[i]);
	}
	system("cls");
	
//	Displaying Array
	printf("Given Array:  ");
	for(i=0; i < size; i++) {
		printf("%d  ",arr[i]);
	}
	
	printf("\n\nEnter the element to be searched:  ");
	scanf("%d",&element);
	
	for(i=0; i < size; i++) {
		if(arr[i] == element) {
			printf("%d found at position:  %d",element, (i+1));
			return;
		}
	}
	printf("%d not found in this Array!",element);
	
}
